#!/usr/bin/env bash
set -euo pipefail

SRC="/home17/jepp/Quest Games"
B2_DEST="b2:quest-archive
TMP="/home17/jepp/ziptmp"
JOBS=4
MAX_RETRIES=3
LOG="$HOME/b2_upload_$(date +%Y%m%d-%H%M%S).log"
SUMMARY_MOVED=0
SUMMARY_SKIPPED=0
SUMMARY_FAILED=0
COUNTERS_DIR="$TMP/counters"

mkdir -p "$TMP" "$COUNTERS_DIR"

# Cleanup tmp on exit (handles crashes too)
trap 'rm -rf "$TMP"' EXIT

log() { echo "[$(date '+%H:%M:%S')] [$1] $2" >> "$LOG"; echo "[$(date '+%H:%M:%S')] [$1] $2"; }

process() {
    game="$1"
    name=$(basename "$game")
    zipfile="$TMP/$name.zip"
    job_log="$LOG.$name"

    # Check if already uploaded to B2
    if rclone lsf "$B2_DEST/$name.zip" >> "$job_log" 2>&1; then
        echo "[$(date '+%H:%M:%S')] [SKIP] $name — already in B2" | tee -a "$LOG"
        touch "$COUNTERS_DIR/skipped_$$"
        return 0
    fi

    # Zip with retries
    for attempt in $(seq 1 $MAX_RETRIES); do
        echo "[$(date '+%H:%M:%S')] [ZIP] $name (attempt $attempt)" | tee -a "$LOG"
        if zip -0 -r "$zipfile" "$game" >> "$job_log" 2>&1; then
            echo "[$(date '+%H:%M:%S')] [ZIP OK] $name" | tee -a "$LOG"
            break
        else
            rm -f "$zipfile"
            if [[ $attempt -eq $MAX_RETRIES ]]; then
                echo "[$(date '+%H:%M:%S')] [FAIL] $name — zip failed after $MAX_RETRIES attempts" | tee -a "$LOG"
                touch "$COUNTERS_DIR/failed_$$"
                return 1
            fi
            sleep 5
        fi
    done

    # Upload with retries
    for attempt in $(seq 1 $MAX_RETRIES); do
        echo "[$(date '+%H:%M:%S')] [UPLOAD] $name (attempt $attempt)" | tee -a "$LOG"
        if rclone copy "$zipfile" "$B2_DEST" \
            --transfers 8 \
            --retries 3 \
            --low-level-retries 10 \
            -P >> "$job_log" 2>&1; then
            echo "[$(date '+%H:%M:%S')] [DONE] $name" | tee -a "$LOG"
            rm -f "$zipfile"
            touch "$COUNTERS_DIR/moved_$$"
            return 0
        else
            if [[ $attempt -eq $MAX_RETRIES ]]; then
                echo "[$(date '+%H:%M:%S')] [FAIL] $name — upload failed after $MAX_RETRIES attempts" | tee -a "$LOG"
                rm -f "$zipfile"
                touch "$COUNTERS_DIR/failed_$$"
                return 1
            fi
            sleep 10
        fi
    done
}

export -f process
export TMP B2_DEST MAX_RETRIES LOG COUNTERS_DIR

echo "[$(date '+%H:%M:%S')] Starting B2 upload — Source: $SRC" | tee -a "$LOG"
echo "[$(date '+%H:%M:%S')] Log: $LOG" | tee -a "$LOG"

find "$SRC" -mindepth 1 -maxdepth 1 -type d -print0 | \
    xargs -0 -P "$JOBS" -I{} bash -c 'process "$@"' _ {}

# Count results from flag files
moved=$(ls "$COUNTERS_DIR"/moved_* 2>/dev/null | wc -l)
skipped=$(ls "$COUNTERS_DIR"/skipped_* 2>/dev/null | wc -l)
failed=$(ls "$COUNTERS_DIR"/failed_* 2>/dev/null | wc -l)

echo ""
echo "─────────────────────────────"
echo "  Uploaded : $moved"
echo "  Skipped  : $skipped (already in B2)"
echo "  Failed   : $failed"
echo "  Log      : $LOG"
echo "─────────────────────────────"

[[ $failed -gt 0 ]] && exit 1
exit 0
EOF

#!/usr/bin/env bash
set -euo pipefail

# ─── CONFIGURATION ──────────────────────────────────────────────────────────
GAMES_DIR="/home17/jepp/Quest Games Clean"
B2_REMOTE="b2:quest-archive"          # rclone remote name and bucket
B2_PATH=""                             # leave empty for root, or "games/"
JOBS=4                                 # parallel zip+upload processes
BW_LIMIT="100M"                        # rclone bandwidth limit
MAX_RETRIES=3
API_BASE="https://your-server.com"    # CHANGE THIS
API_KEY="your-api-key-here"           # CHANGE THIS
LOG_DIR="$HOME"
LOG_FILE="$LOG_DIR/upload-$(date +%Y%m%d-%H%M%S).log"
TMP_COUNTERS="/tmp/b2_upload_counters_$$"
mkdir -p "$TMP_COUNTERS"

trap 'rm -rf "$TMP_COUNTERS"' EXIT

# ─── HELPERS ─────────────────────────────────────────────────────────────────
log()   { echo "[$(date '+%H:%M:%S')] $*" | tee -a "$LOG_FILE"; }
info()  { echo -e "\033[0;36m[INFO]\033[0m  $*"; log "[INFO]  $*"; }
ok()    { echo -e "\033[0;32m[UPLOAD]\033[0m $*"; log "[UPLOAD] $*"; }
skip()  { echo -e "\033[1;33m[SKIP]\033[0m  $*"; log "[SKIP]  $*"; }
err()   { echo -e "\033[0;31m[ERROR]\033[0m $*" >&2; log "[ERROR] $*"; }
die()   { err "$*"; exit 1; }

# ─── hashId function (mirrors your backend's hashId) ────────────────────────
# Uses HMAC-SHA256 with HASH_SECRET from your .env; pass secret as env var
# If you don't set HASH_SECRET, it falls back to plain SHA256 of the name.
compute_hash() {
    local name="$1"
    if [[ -n "${HASH_SECRET:-}" ]]; then
        echo -n "$name" | openssl dgst -sha256 -hmac "$HASH_SECRET" -hex | cut -d' ' -f2 | cut -c1-32
    else
        echo -n "$name" | sha256sum | cut -c1-32
    fi
}

# ─── Register upload with your backend API ──────────────────────────────────
register_upload() {
    local hash="$1"
    local key="$2"   # e.g., "GameName.zip"
    if [[ -z "$API_BASE" || "$API_BASE" == "https://your-server.com" ]]; then
        info "Skipping API registration (no API_BASE configured)"
        return 0
    fi
    local status_code
    status_code=$(curl -s -o /dev/null -w "%{http_code}" -X POST "$API_BASE/api/internal/register-upload" \
        -H "Authorization: Bearer $API_KEY" \
        -H "Content-Type: application/json" \
        -d "{\"hash\":\"$hash\",\"key\":\"$key\"}")
    if [[ "$status_code" -eq 200 || "$status_code" -eq 204 ]]; then
        info "Registered: $hash → $key"
    else
        err "API registration failed (HTTP $status_code) for $key"
        return 1
    fi
}

# ─── Process a single game folder ───────────────────────────────────────────
process_game() {
    local folder="$1"
    local name=$(basename "$folder")
    local remote_key="${B2_PATH}${name}.zip"
    local hash=""

    # Skip if already exists in B2 (optional – rclone rcat would overwrite)
    if rclone lsf "${B2_REMOTE}/${remote_key}" >/dev/null 2>&1; then
        skip "$name (already in B2)"
        touch "$TMP_COUNTERS/skipped"
        return 0
    fi

    # Compute hash before upload
    hash=$(compute_hash "$name")
    info "Processing: $name (hash: $hash)"

    # Streaming zip + upload (no temp file)
    for attempt in $(seq 1 $MAX_RETRIES); do
        info "Attempt $attempt for $name"
        if (
            cd "$folder" && \
            zip -0 -r - . | \
            rclone rcat "${B2_REMOTE}/${remote_key}" \
                --bwlimit "$BW_LIMIT" \
                --retries 3 \
                --low-level-retries 5 \
                --checksum \
                --progress \
                2>>"$LOG_FILE"
        ); then
            ok "$name -> ${remote_key}"
            # Register with backend
            register_upload "$hash" "$remote_key" || true
            touch "$TMP_COUNTERS/uploaded"
            return 0
        else
            err "Attempt $attempt failed for $name"
            sleep 5
        fi
    done

    err "FAILED after $MAX_RETRIES attempts: $name"
    touch "$TMP_COUNTERS/failed"
    return 1
}
export -f process_game compute_hash register_upload
export B2_REMOTE B2_PATH BW_LIMIT MAX_RETRIES API_BASE API_KEY LOG_FILE TMP_COUNTERS

info "Log: $LOG_FILE"

echo "  Uploaded : $uploaded"

echo "  Log      : $LOG_FILE"

[[ $failed -eq 0 ]] || exit 1
exit 0
